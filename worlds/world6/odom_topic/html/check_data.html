bag = rosbag.Bag('ottobock_2021-03-03-09-03-16.bag')
# get all poses and time
time_pose = np.array([])
pose_x = np.array([])
pose_y = np.array([])
for topic, msg, t in bag.read_messages(topics=['/odom']):
    t = rospy.Time.to_sec(t) # convert to float
    time_pose = np.append(time_pose, t)
    pose_x = np.append(pose_x, msg.pose.pose.position.x)
    pose_y = np.append(pose_y, msg.pose.pose.position.y)


plt.figure(figsize=(11.69,8.27))
plt.axis('equal')
plt.plot(pose_x, pose_y) 
plt.legend(["odometry_map"])
plt.xlabel('Pos. x in $m$')
plt.ylabel('Pos. y in $m$')
plt.show()
